source("./SDTM_RS_LCHPU_data.r")
 
 
#==============================================================================;
#Read and process input datasets;
#==============================================================================;
 
rs01 <- chpu
dm01 <- dm
 
#==============================================================================;
#Create required variables in rs;
#==============================================================================;
rs02 <- bind_rows(
 mutate(rs01, rstestcd = "CPS0101", rsorres = word(chpu_1, 2, sep=":"), rsstresc = word(chpu_1, 1, sep=":")),
 mutate(rs01, rstestcd = "CPS0102", rsorres = word(chpu_2, 2, sep=":"), rsstresc = word(chpu_2, 1, sep=":")),
 mutate(rs01, rstestcd = "CPS0103", rsorres = word(chpu_3, 2, sep=":"), rsstresc = word(chpu_3, 1, sep=":")),
 mutate(rs01, rstestcd = "CPS0104", rsorres = word(chpu_4, 2, sep=":"), rsstresc = word(chpu_4, 1, sep=":")),
 mutate(rs01, rstestcd = "CPS0105A", rsorres = word(chpu_5, 2, sep=":"), rsstresc = word(chpu_5, 1, sep=":")),
) %>%
 mutate(
 studyid = "MYCSG",
 domain = "RS",
 usubjid = paste(studyid, subject, sep = "-"),
 rsdtc = ifelse(rsdat_raw != "", format(as.Date(rsdat_raw, format = "%d %b %Y"), "%Y-%m-%d"), ""),
 rscat = "CHILD-PUGH CLASSIFICATION",
 visitnum = folderseq,
 visit = foldername,
 rsstresn = suppressWarnings(as.numeric(rsstresc)),
 rsorresu = "",
 rsstresu = ""
 )
 
 
#==============================================================================;
#Create --LOBXFL;
#==============================================================================;
 
#------------------------------------------------------------------------------;
#Fetch RFXSTDTC into rs data;
#------------------------------------------------------------------------------;
 
rs03 <- rs02 %>%
 arrange(studyid, usubjid) %>%
 left_join(dm01 %>% select(studyid, usubjid, rfstdtc, rfxstdtc), by = c("studyid", "usubjid"))
 
 
#------------------------------------------------------------------------------;
#Filter qualifying records;
#------------------------------------------------------------------------------;
base01 <- rs03 %>%
 filter(rsdtc != "", rsdtc <= rfxstdtc, rsorres != "")
 
#------------------------------------------------------------------------------;
#Pick latest record;
#------------------------------------------------------------------------------;
base02 <- base01 %>%
 arrange(studyid, usubjid, rstestcd, rsdtc) %>%
 group_by(studyid, usubjid, rstestcd) %>%
 filter(row_number() == n()) %>%
 ungroup() %>%
 select(studyid, usubjid, rstestcd, rsdtc)
 
#------------------------------------------------------------------------------;
#populate --lobxfl on the timepoint identified above;
#------------------------------------------------------------------------------;
 
rs04 <- rs03 %>%
 left_join(base02 %>% mutate(rslobxfl = "Y"),
 by = c("studyid", "usubjid", "rstestcd", "rsdtc")) %>%
 mutate(rslobxfl = if_else(!is.na(rslobxfl), rslobxfl, ""))
 
 
 
#==============================================================================;
#Derive other dependent variables;
#==============================================================================;
rs05 <- rs04 %>%
 mutate(
 rstest = case_when(
 rstestcd == 'CPS0101' ~ 'CPS01-Encephalopathy Grade',
 rstestcd == 'CPS0102' ~ 'CPS01-Ascites',
 rstestcd == 'CPS0103' ~ 'CPS01-Serum Bilirubin',
 rstestcd == 'CPS0104' ~ 'CPS01-Serum Albumin',
 rstestcd == 'CPS0105A' ~ 'CPS01-PT, Sec Prolonged',
 TRUE ~ ""
 ),
 rsdt = as.Date(rsdtc, format = "%Y-%m-%d"),
 rfstdt = as.Date(rfstdtc, format = "%Y-%m-%d"),
 rsdy = if_else(!is.na(rsdt) & !is.na(rfstdt), rsdt - rfstdt + (rsdt >= rfstdt), NA),
 rsdy=as.numeric(rsdy),
 rsdt = format(rsdt, "%Y-%m-%d"),
 rfstdt = format(rfstdt, "%Y-%m-%d")
 )
 
 
#==============================================================================;
#Create --SEQ variable;
#==============================================================================;
rs06 <- rs05 %>%
 arrange(studyid, usubjid, rstestcd, rsdtc) %>%
 group_by(studyid, usubjid) %>%
 mutate(rsseq = row_number()) %>%
 ungroup() %>%
 rename_all(toupper)
 
 
#==============================================================================;
#keep only required variables ;
#==============================================================================;
varlist <- c("STUDYID", "DOMAIN", "USUBJID", "RSSEQ", "RSTESTCD", "RSTEST",
 "RSCAT", "RSORRES", "RSORRESU", "RSSTRESC", "RSSTRESN",
 "RSSTRESU", "RSLOBXFL", "VISITNUM", "VISIT", "RSDTC", "RSDY")
 
rs <- rs06 %>%
 select(all_of(varlist))
